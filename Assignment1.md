- [ ] 1.7.2 Algorithm Workbench, Choose any 1 question 
  3. Write a function that receives an integer. The function must return a string containing the
binary representation of the integer.
Using C++ :

#include <iostream>
#include <cstdlib>

using namespace std;

void decToBinary(int);

int main() {
	int value;

	cout << "Enter a non-negative integer value: ";
	cin >> value;
	cout << "Decimal " << value << " = ";
	decToBinary(value);
	cout << " in binary.\n";
	return 0;
}

void decToBinary(int num) {
	if (num == 0) {
		cout << "0";
	}
	else if (num < 0) {
		cout << "Invalid Entry";
	}
	else if (num > 0) {
		decToBinary(num / 2);
		cout << num % 2;
	}
}
